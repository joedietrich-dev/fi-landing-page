      function solutionRefactor(roman){
        const values = {
          'M': 1000,
          'D': 500,
          'C': 100,
          'L': 50,
          'X': 10,
          'V': 5,
          'I': 1,
        }
        let total = 0;
        for (let i = 0; i < roman.length; i++) {
          // If this letter's value is less than the next letter's, we subtract it
          const curr = values[roman[i]];
          const next = values[roman[i + 1]];
          total += (curr < next) ? -curr : curr;
        }
        return total;
      }
